// In clean installed host, clone kubespray:
git clone https://github.com/kubernetes-sigs/kubespray.git

// Switch to the tag which supports the k8s version you want to upgrade (e.g. 1.27)
cd kubespray
git checkout v2.23.3

// Look at the version of ansible, kube and calico version (also in nodes)
cat playbooks/ansible_version.yml
grep kube_version inventory/sample/group_vars/k8s_cluster/k8s-cluster.yml
grep -ri calico_version .

kubectl get pods -n kube-system -l k8s-app=calico-node -o jsonpath='{.items[0].spec.containers[0].image}'

// Indentify from AI which version of python must stand both on ansible and cluster nodes according to ansible version
// (e.g. i want to upgrade my k8s cluster from 1.26 to 1.27 with kubespray.
// tag v2.23.3 supports upgrading to v1.27.7 and requires ansible min 2.14.0 and max 2.15.0.
// which python version do i have to install on ansible host  and on the cluster nodes? currenly it is 3.8.10 on the nodes.

// Install appropriate version on the ansible node, in this case 3.9
sudo apt install software-properties-common -y
sudo add-apt-repository ppa:deadsnakes/ppa -y
sudo apt update && sudo apt install python3.9 python3.9-venv sshpass -y
python3.9 --version

// Create python env
cd ../
python3.9 -m venv ansible-2.14-venv
source ansible-2.14-venv/bin/activate

// Install the requirements and check ansible version
cd kubespray
pip install -r requirements.txt
ansible --version

// Run pre-upgrade commands
cp -rfp inventory/sample inventory/mycluster
declare -a IPS=(10.0.6.50 10.0.6.51 10.0.6.52 10.0.6.53 10.0.6.54)
CONFIG_FILE=inventory/mycluster/hosts.yaml python3 contrib/inventory_builder/inventory.py ${IPS[@]}

// Edit hostnames if necessary
vi inventory/mycluster/hosts.yaml

// Create access file
vi access_to_servers.yml

// Check if hosts are reachable
ansible all -i inventory/mycluster/hosts.yaml -m ping --extra-vars='@access_to_servers.yml'

// Run the upgrade playbook
ansible-playbook -i inventory/mycluster/hosts.yaml --become --become-user=root --extra-vars='@access_to_servers.yml' upgrade-cluster.yml
